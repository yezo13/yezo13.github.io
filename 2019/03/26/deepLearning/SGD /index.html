#定义代价函数及梯度下降函数

#代价函数 计算误差
#对预测值和y作差，平方后求和；
def costFunc(X,Y,theta):
    inner = np.power((X*theta.T)-Y,2)  
    return np.sum(inner)/(2*len(X))  


    

#梯度下降函数

#X,Y,theta：系数集，lr:学习率，iters：迭代次数

def SgradientDescent(X,Y,Xtest,Ytest,theta,lr,iters):
    #初始化
    temp = np.mat(np.zeros(theta.shape))
    trainCost = []
    testCost = []
    index = []
    allRan = []
    thetaNums = int(theta.shape[1]) #系数的个数
    #迭代过程
    for i in range(iters):
        tempi = np.random.randint(0,len(X)) #随机取一个index 作为这次迭代的参数
       # allRan.append(tempi)
        error = np.dot(X[tempi],theta.T) - Y[tempi] 
        ##error = (X*theta.T-Y) #计算误差
        for j in range(thetaNums): #三个系数
            #np.multiply：数组和矩阵对应位置相乘，输出与相乘数组/矩阵的大小一致
            ##derivativeInner = np.multiply(error,X[:,j]) #求导数
            derivativeInner = np.dot(error,X[tempi,j])
            
            #得到当前的参数
            ##temp[0,j] = theta[0,j] - (lr*np.sum(derivativeInner)/len(X))
            if(i == 0):
                print("temp",type(temp[0,j]))
                print("theta",type(theta[0,j]))
                print("lr*de",type(lr*np.sum(derivativeInner)))
                pass
            
            temp[0,j] = theta[0,j] - lr*np.sum(derivativeInner)
            
        theta = temp #将参数更新
        if(i % 10000 == 0):
            index.append(i);
            trainCost.append(costFunc(X,Y,theta))
            testCost.append(costFunc(Xtest,Ytest,theta))
    
    return theta,trainCost,testCost,index
